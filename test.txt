from PyQt5.QtWidgets import QApplication, QMainWindow, QMenuBar, QAction, QInputDialog
import sys

class MyWindow(QMainWindow):
    def __init__(self):
        super().__init__()

        self.menu_bar = QMenuBar()
        self.file_menu = self.menu_bar.addMenu("File")
        self.exit_action = QAction("Exit")
        self.exit_action.triggered.connect(self.close)
        self.file_menu.addAction(self.exit_action)

        self.input_menu = self.menu_bar.addMenu("Input")
        self.input_action = QAction("Input Text")
        self.input_action.triggered.connect(self.input_text)
        self.input_menu.addAction(self.input_action)

        self.setMenuBar(self.menu_bar)

    def input_text(self):
        text, ok = QInputDialog.getText(self, "Input Dialog", "Enter text:")
        if ok:
            print(text)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = MyWindow()
    window.show()
    sys.exit(app.exec_())
