def get_crossroad(node, N_sizes, N_size_big, mode):  # расчет номера перекрестка по номеру ячейки
	N_size=sum(N_sizes)
	
	Sum=0
	sz=len(N_sizes)
	block_node=0
	node_tmp=node%N_size

	for i in range(0, sz): #это не меняем
		Sum=Sum+N_sizes[i]
		if(node_tmp<Sum): #<=
			block_node=i
			break
	
	cross_row=0 #нумерация с нуля
	if(mode==2):
		cross_row=int(node/(2*N_size))
		return(cross_row*N_size_big+node_tmp+block_node+1)

	if(mode==1):
		cross_row=int(node/(2*N_size))
		return(cross_row*N_size_big+node_tmp+block_node+2)


N_sizes=[3, 2, 3, 2, 2]
mode=1
if(mode==1):
	N_size=sum(N_sizes)+len(N_sizes)+1
if(mode==2):
	N_size=sum(N_sizes) + len(N_sizes) - 1


node=69
print(get_crossroad(node, N_sizes, N_size, mode))
