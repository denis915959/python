path=[1, 2, 3, 3, 3, 2, 1]

def get_path_nodes(path): #строит маршрут, состоящий из номеров перекрестков 
	sz=len(path)
	path_nodes=[]
	tmp=path[0]
	path_nodes.append(tmp)
	for i in range(1, sz):
		if(path[i]!=tmp):
			tmp=path[i]
			path_nodes.append(tmp)
	return(path_nodes)

print(get_path_nodes(path))
